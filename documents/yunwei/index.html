自动安装Nginx

#!/bin/bash #操作系统: CentOS #add repository releasever=`rpm -qi centos-release | grep Version | awk -F ':' '{print $2}'| awk -F ' ' '{print $1}'`; basearch=`arch`; echo "[nginx] name=nginx repo baseurl=http://nginx.org/packages/centos/$releasever/$basearch/ gpgcheck=0 enabled=1">/etc/yum.repos.d/nginx.repo #install nginx yum install nginx #start nginx service nginx start
